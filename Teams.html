    <script>

        const teams = [

            { id: 'red', name: 'Red Team', count: 5, members: [], color: '#ff0000', startAngle: 0, endAngle: 120 },

            { id: 'blue', name: 'Blue Team', count: 5, members: [], color: '#0000ff', startAngle: 120, endAngle: 240 },

            { id: 'green', name: 'Green Team', count: 5, members: [], color: '#00ff00', startAngle: 240, endAngle: 360 }

        ];


        let isSpinning = false;


        // Update team names when input changes

        document.getElementById('red-name').addEventListener('input', (e) => {

            document.getElementById('red-name-display').textContent = e.target.value || 'Red Team';

        });

        document.getElementById('blue-name').addEventListener('input', (e) => {

            document.getElementById('blue-name-display').textContent = e.target.value || 'Blue Team';

        });

        document.getElementById('green-name').addEventListener('input', (e) => {

            document.getElementById('green-name-display').textContent = e.target.value || 'Green Team';

        });


        function spinWheel() {

            if (isSpinning) return;

            

            const nameInput = document.getElementById('name');

            const name = nameInput.value.trim();

            if (!name) {

                alert('Please enter your name!');

                return;

            }


            const availableTeams = teams.filter(t => t.count > 0);

            if (availableTeams.length === 0) {

                alert('All teams are full!');

                return;

            }


            isSpinning = true;

            document.getElementById('spinButton').disabled = true;


            // Play spinning sound

            const spinSound = document.getElementById('spinSound');

            spinSound.currentTime = 0;

            spinSound.play();


            // Select random team

            const selectedTeamIndex = Math.floor(Math.random() * availableTeams.length);

            const selectedTeam = availableTeams[selectedTeamIndex];


            // Calculate rotation (5 full rounds + random angle within the team's sector)

            const fullRotations = 1800; // 5 full rotations (360 * 5)

            const randomAngle = Math.random() * (selectedTeam.endAngle - selectedTeam.startAngle) + selectedTeam.startAngle;

            const finalRotation = fullRotations + randomAngle;


            // Update pointer color to match the team

            const pointer = document.getElementById('pointer');

            pointer.style.backgroundColor = selectedTeam.color;


            // Spin the wheel

            const wheel = document.getElementById('wheel');

            wheel.style.transform = `rotate(${finalRotation}deg)`;


            setTimeout(() => {

                // Play winning sound

                const winSound = document.getElementById('winSound');

                winSound.currentTime = 0;

                winSound.play();


                // Update team data

                selectedTeam.count--;

                selectedTeam.members.push(name);


                // Update UI

                updateTeamDisplay(selectedTeam);

                nameInput.value = '';

                isSpinning = false;

                document.getElementById('spinButton').disabled = false;

            }, 3000);

        }


        function updateTeamDisplay(team) {

            // Update count display

            document.getElementById(`${team.id}-count`).textContent = team.count;


            // Update member list

            const list = document.getElementById(`${team.id}-team`);

            const listItem = document.createElement('li');

            listItem.textContent = team.members[team.members.length - 1];

            list.appendChild(listItem);

        }
